A Mono reacts as soon as the System starts and produces exactly one value (the value set by the constructor).

fn main() {
    let handle = System::init().unwrap();
    let gen = Mono::new(0u8).unwrap();

    System::start(gen.map(|i| {
        assert_eq!(i, 0u8);
        handle.send(SystemEvent::Stop) 
    }));
}
